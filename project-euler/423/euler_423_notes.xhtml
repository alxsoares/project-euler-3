\[ x = 6 \\ x_{i=1,n} = 5^{n-1} \cdot x \\ x_{i=2,n=2} = x_{i=1,n=1} = 5^n = 5^1 \cdot x \\ x_{i=i',n=n'} = x_{i=i'-1,n=n'-1} + 5*x_{i=i'-1,n=n'} \\ x_{i=2,n=3} = 5^1 \cdot 5 + 5^2 = 2 \cdot 5^2 \cdot x \\ x_{i=2,n=4} = 2 \cdot 5^2 \cdot 5 + 5^3 = 3 \cdot 5^3 \cdot x \\ x_{n,i=2} = (n-1) \cdot 5^{n-2} \cdot x \\ \]
